  2. Program NKINV;
  3. Const
  4.   maxN =60000;
  5.   maxV =60000;
  6. Var
  7.   n,ans :LongInt;
  8.   A :Array[0..maxN] of LongInt;
  9.   BIT :Array[0..maxV] of LongInt;
  10.  
  11.   procedure Enter;
  12.   var
  13.     i :LongInt;
  14.   begin
  15.     Read(n);
  16.     for i:=1 to n do Read(A[i]);
  17.   end;
  18.  
  19.   procedure UpdateBIT(i :LongInt);
  20.   begin
  21.     while (i<=maxV) do
  22.       begin
  23.         Inc(BIT[i]);
  24.         Inc(i,i and (-i));
  25.       end;
  26.   end;
  27.  
  28.   function SearchBIT(i :LongInt) :LongInt;
  29.   var
  30.     res :LongInt;
  31.   begin
  32.     res:=0;
  33.     while (i>0) do
  34.       begin
  35.         Inc(res,BIT[i]);
  36.         Dec(i,i and (-i));
  37.       end;
  38.     Exit(res);
  39.   end;
  40.  
  41.   procedure SolveBIT;
  42.   var
  43.     i :LongInt;
  44.   begin
  45.     FillChar(BIT,SizeOf(BIT),0);
  46.     ans:=0;
  47.     for i:=n downto 1 do
  48.       begin
  49.         Inc(ans,SearchBIT(A[i]-1));
  50.         UpdateBIT(A[i]);
  51.       end;
  52.   end;
  53.  
  54. Begin
  55.   Assign(Input,''); Reset(Input);
  56.   Assign(Output,''); Rewrite(Output);
  57.   Enter;
  58.   SolveBIT;
  59.   Write(ans);
  60.   Close(Input); Close(Output);
  61. End.
