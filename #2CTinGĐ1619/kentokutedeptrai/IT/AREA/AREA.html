  1. uses math;
  2. const
  3.         tfi     =       '';
  4.         tfo     =       '';
  5.         nmax    =       30001;
  6. type
  7.         ban = record
  8.           gt,up,down:longint;
  9.           ok:boolean;
  10.         end;
  11.  
  12.         node = record
  13.            sl,dt:longint;
  14.         end;
  15. var
  16.         fi,fo:text;
  17.         n,m,top,maxy:longint;
  18.         st:array[0..nmax] of ban;
  19.         res:longint;
  20.         t:Array[0..10*nmax] of node;
  21. procedure mo;
  22. var i,j:longint;
  23. begin
  24.         assign(fi,tfi);reset(fi);
  25.         assign(fo,tfo);rewrite(fo);
  26. end;
  27. procedure nhap;
  28. var i,j,x1,y1,x2,y2:longint;
  29. begin
  30.         read(fi,n);
  31.         for i:=1 to n do
  32.           begin
  33.            read(fi,x1,y1,x2,y2);
  34.            inc(top);
  35.            st[top].gt:=x1;
  36.            st[top].up:=y2;
  37.            st[top].down:=y1;
  38.            st[top].ok:=true;
  39.            inc(top);
  40.            st[top].gt:=x2;
  41.            st[top].up:=y2;
  42.            st[top].down:=y1;
  43.            st[top].ok:=false;
  44.            maxy:=max(maxy,max(y1,y2));
  45.           end;
  46. end;
  47. procedure updateopen(i,lo,hi,u,v:longint);
  48. var mid:longint;
  49. begin
  50.         if lo+1>=hi then
  51.           begin
  52.             if (u<=lo)and(hi<=v) then
  53.               begin
  54.                 inc(t[i].sl);
  55.                 t[i].dt:=1;
  56.               end;
  57.             exit;
  58.           end;
  59.         if (v<lo)or(hi<u) then exit;
  60.         if (u<=lo)and(hi<=v) then
  61.           begin
  62.             inc(t[i].sl);
  63.             t[i].dt:=hi-lo;
  64.             exit;
  65.           end;
  66.         mid:=(lo+hi)shr 1;
  67.         updateopen(2*i,lo,mid,u,v);
  68.         updateopen(2*i+1,mid,hi,u,v);
  69.         if t[i].sl=0 then t[i].dt:=t[2*i].dt+t[2*i+1].dt;
  70. end;
  71. procedure updateclose(i,lo,hi,u,v:longint);
  72. var mid:longint;
  73. begin
  74.         if lo+1>=hi then
  75.           begin
  76.            if (u<=lo)and(hi<=v) then
  77.              begin
  78.                dec(t[i].sl);
  79.                if t[i].sl=0 then t[i].dt:=0;
  80.              end;
  81.            exit;
  82.           end;
  83.         if (v<lo)or(hi<u) then exit;
  84.         if (u<=lo)and(hi<=v) then
  85.           begin
  86.             dec(t[i].sl);
  87.             if t[i].sl=0 then
  88.                t[i].dt:=t[2*i].dt+t[2*i+1].dt;
  89.             exit;
  90.           end;
  91.         mid:=(lo+hi)shr 1;
  92.         updateclose(2*i,lo,mid,u,v);
  93.         updateclose(2*i+1,mid,hi,u,v);
  94.         if t[i].sl=0 then t[i].dt:=t[2*i].dt+t[2*i+1].dt;
  95. end;
  96. procedure doicho(var i,j:ban);
  97. var tg:ban;
  98. begin
  99.         tg:=i;
  100.         i:=j;
  101.         j:=tg;
  102. end;
  103. procedure sort(l,r:longint);
  104. var i,j,mid:longint;
  105. begin
  106.         i:=l;
  107.         j:=r;
  108.         mid:=st[(l+r)shr 1].gt;
  109.         repeat
  110.           while st[i].gt<mid do inc(i);
  111.           while st[j].gt>mid do dec(j);
  112.           if i<=j then
  113.             begin
  114.               doicho(st[i],st[j]);
  115.               inc(i);
  116.               dec(j);
  117.             end;
  118.         until i>j;
  119.         if l<j then sort(l,j);
  120.         if i<r then sort(i,r);
  121. end;
  122. procedure khoitao;
  123. var i,j:longint;
  124. begin
  125.         sort(1,top);
  126.         updateopen(1,0,maxy,st[1].down,st[1].up);
  127. end;
  128. procedure xuly;
  129. var i,j:longint;
  130. begin
  131.         for i:=2 to top do
  132.           begin
  133.             res:=res+(st[i].gt-st[i-1].gt)*t[1].dt;
  134.             if st[i].ok=true then updateopen(1,0,maxy,st[i].down,st[i].up)
  135.             else updateclose(1,0,maxy,st[i].down,st[i].up);
  136.           end;
  137. end;

  139. var i,j:longint;
  140. begin
  141.         writeln(fo,res);
  142. end;
